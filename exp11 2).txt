//2.	Write a program to apply left shift and right shift operator.

#include <stdio.h>

int main() {
    int num;

    printf("Enter a number: ");
    scanf("%d", &num);

    printf("\nLeft shift (num << 1) = %d", num << 1);
    printf("\nRight shift (num >> 1) = %d", num >> 1);

    return 0;
}
