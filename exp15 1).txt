1) WAP to create a shared library for performing arithmetic functions

FILE 1 : arith_shared.h

int add(int, int);
int sub(int, int);
int mul(int, int);
int divide(int, int);


FILE 2 : arith_shared.c

#include "arith_shared.h"

int add(int a, int b)
{
    return a + b;
}

int sub(int a, int b)
{
    return a - b;
}

int mul(int a, int b)
{
    return a * b;
}

int divide(int a, int b)
{
    return a / b;
}
